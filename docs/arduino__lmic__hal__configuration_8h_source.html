41//
42// for legacy reasons, we need a plain-old-data C-like
43// structure that defines the "pin mapping" for the
44// common pins. Many clients initialize an instance of
45// this structure using named-field initialization.
46//
47// Be careful of alignment below.
48struct HalPinmap_t
49{
50 // Use this for any unused pins.
51 static constexpr uint8_t UNUSED_PIN = 0xff;
52 static constexpr int NUM_DIO = 3;
53 // for backward compatibility...
54 static constexpr uint8_t LMIC_UNUSED_PIN = UNUSED_PIN;
55
56 /* the contents */
57 uint8_t nss; // byte 0: pin for select
58 uint8_t rxtx; // byte 1: pin for rx/tx control
59 uint8_t rst; // byte 2: pin for reset
60 uint8_t dio[NUM_DIO]; // bytes 3..5: pins for DIO0, DOI1, DIO2
61 // true if we must set rxtx for rx_active, false for tx_active
62 uint8_t rxtx_rx_active; // byte 6: polarity of rxtx active
63 int8_t rssi_cal; // byte 7: cal in dB -- added to RSSI
64 // measured prior to decision.
65 // Must include noise guardband!
66 uint32_t spi_freq; // bytes 8..11: SPI freq in Hz.
67
68 // optional pointer to configuration object (bytes 12..15)
69 HalConfiguration_t *pConfig;
70};
