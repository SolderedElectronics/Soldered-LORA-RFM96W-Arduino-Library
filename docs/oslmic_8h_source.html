260// ======================================================================
261// Table support
262// These macros for defining a table of constants and retrieving values
263// from it makes it easier for other platforms (like AVR) to optimize
264// table accesses.
265// Use CONST_TABLE() whenever declaring or defining a table, and
266// TABLE_GET_xx whenever accessing its values. The actual name of the
267// declared variable will be modified to prevent accidental direct
268// access. The accessor macros forward to an inline function to allow
269// proper type checking of the array element type.
270
271// Helper to add a prefix to the table name
272#define RESOLVE_TABLE(table) constant_table_##table
273
274// get number of entries in table
275#define LENOF_TABLE(table) (sizeof(RESOLVE_TABLE(table)) / sizeof(RESOLVE_TABLE(table)[0]))
276
277// Accessors for table elements
278#define TABLE_GET_U1(table, index) table_get_u1(RESOLVE_TABLE(table), index)
279#define TABLE_GET_S1(table, index) table_get_s1(RESOLVE_TABLE(table), index)
280#define TABLE_GET_U2(table, index) table_get_u2(RESOLVE_TABLE(table), index)
281#define TABLE_GET_S2(table, index) table_get_s2(RESOLVE_TABLE(table), index)
282#define TABLE_GET_U4(table, index) table_get_u4(RESOLVE_TABLE(table), index)
283#define TABLE_GET_S4(table, index) table_get_s4(RESOLVE_TABLE(table), index)
284#define TABLE_GET_OSTIME(table, index) table_get_ostime(RESOLVE_TABLE(table), index)
285#define TABLE_GET_U1_TWODIM(table, index1, index2) table_get_u1(RESOLVE_TABLE(table)[index1], index2)
286
287#if defined(__AVR__)
288#include <avr/pgmspace.h>
289// Macro to define the getter functions. This loads data from
290// progmem using pgm_read_xx, or accesses memory directly when the
291// index is a constant so gcc can optimize it away;
292#define TABLE_GETTER(postfix, type, pgm_type) \
293 static inline type table_get##postfix(const type *table, size_t index) \
294 { \
295 if (__builtin_constant_p(table[index])) \
296 return table[index]; \
297 return pgm_read_##pgm_type(&table[index]); \
298 }
299

301TABLE_GETTER(_s1, s1_t, byte);
302TABLE_GETTER(_u2, u2_t, word);
303TABLE_GETTER(_s2, s2_t, word);
304TABLE_GETTER(_u4, u4_t, dword);
